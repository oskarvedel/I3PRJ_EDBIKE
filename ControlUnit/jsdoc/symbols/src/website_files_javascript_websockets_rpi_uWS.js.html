  2 var wsUri = "ws://10.9.8.2:3000/";
  3 var output;
  4 var counter = 0;
  5 
  6 /**
  7  * Initialiserer uWebSockets
  8  */
  9 function init() {
 10     output = document.getElementById("output");
 11     websocket = new WebSocket(wsUri);
 12     initialiseFunctions();
 13 }
 14 /**
 15  * Initialiserer uWebSockets' basisfunktioner
 16  */
 17 function initialiseFunctions() {
 18 
 19     websocket.onopen = function (evt) {
 20         onOpen(evt)
 21     };
 22 
 23     websocket.onclose = function (evt) {
 24         onClose(evt)
 25     };
 26 
 27     websocket.onmessage = function (evt) {
 28         onMessage(evt)
 29     };
 30     websocket.onerror = function (evt) {
 31         onError(evt)
 32     };
 33 }
 34 
 35 /**
 36  * Sender besked til serveren om, at klienten er forbundet
 37  */
 38 function onOpen(evt) {
 39     doSend("EDBIKE Client Connected to Server");
 40 }
 41 
 42 /**
 43  * Sender besked til serveren om, at klienten ikke l√¶ngere er forbundet
 44  */
 45 function onClose(evt) {
 46     doSend("EDBIKE Client Disconnected from Server");
 47 }
 48 
 49 /**
 50  * Sender data til serveren
 51  */
 52 function doSend(Message) {
 53     websocket.send(Message); //sends to server
 54 }
 55 
 56 /**
 57  * Modtager data fra serveren
 58  */
 59 function onMessage(evt) { //receives from server
 60     handleEvent(evt.data);
 61 }
 62 
 63 window.addEventListener("load", init, false);
