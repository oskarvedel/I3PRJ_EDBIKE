 39 /**
 40  * Tager imod et event fra serveren og kalder den funktion, der skal håndtere dette event
 41  * @param {string} event Det EDBike event og data, der er sendt til brugerfladen fra serveren
 42  */
 43 function handleEvent(event) {
 44     var eventType = event.substr(0, 2)
 45     switch (eventType) {
 46         case eventEnums.SERVER_SENT_LIVEDATA:
 47             updateLiveData(event)
 48             break;
 49         case eventEnums.PIN_VALIDATE_RETURNED_CORRECT_PIN:
 50             unlockUI();
 51             break;
 52         case eventEnums.PIN_VALIDATE_RETURNED_WRONG_PIN:
 53             wrongPin(event);
 54             break;
 55         case eventEnums.PIN_VALIDATE_RETURNED_TIMEOUT:
 56             timeOut(event);
 57             break;
 58         case eventEnums.SERVER_SENT_RIDES:
 59             updateRideData(event);
 60             break;
 61         case eventEnums.SERVER_SENT_NUMBER_OF_RIDES:
 62             writeToNumOfRides(event)
 63             break;
 64         case eventEnums.LOCKED:
 65             UIlockedBySlave();
 66             break;
 67         case "gg":
 68             setWheelCirc();
 69             break;
 70         case "dd":
 71             illegalWheelSize();
 72             break;
 73         default:
 74             break;
 75     }
 76 }

126 /**
127  * Tager imod ID'et på en tur, der skal slettes, og sender det til serveren sammen med eventet USER_DELETED_RIDE.
128  * Opdaterer derefter brugerfladen ved at trække alle rides fra serveren.
129  * @param {int} ride ID'et på den tur, der skal slettes
130  */
131 function deleteRide(ride) {
132     doSend(eventEnums.USER_DELETED_RIDE + ride);
133     document.getElementById("rideDataContent").innerHTML = ""; //remove all rides from UI
134     doSend(eventEnums.USER_REQUESTS_RIDES);
135 }

137 /**
138  * Tager imod en tur fra databasen og genererer et collapsible html-element med turdataene.
139  * @param {string} SERVER_SENT_RIDES Den tur, der er sendt fra serveren
140  */
141 function updateRideData(SERVER_SENT_RIDES) {
142     //create variables with ride data
143     var rideNum = SERVER_SENT_RIDES.substr(3, 3);
144     var distance = "<b>Distance:</b> " + SERVER_SENT_RIDES.substr(7, 3) + " km";
145     var topSpeed = "<b>Tophastighed:</b> " + SERVER_SENT_RIDES.substr(11, 3) + " km/h";
146     var avgSpeed = "<b>Gennemsnitlig hastighed:</b> " + SERVER_SENT_RIDES.substr(15, 2) + " km/h";
147     var startTime = "<b>Turen begyndte klokken:</b> " + SERVER_SENT_RIDES.substr(18, 2) + ":" + SERVER_SENT_RIDES.substr(20, 2);
148     var endTime = "<b>Turen sluttede klokken:</b> " + SERVER_SENT_RIDES.substr(23, 2) + ":" + SERVER_SENT_RIDES.substr(25, 2);
149     var date = "<b>Dato:</b> " + SERVER_SENT_RIDES.substr(28, 2) + "/" + SERVER_SENT_RIDES.substr(30, 2) + "/" + SERVER_SENT_RIDES.substr(32, 2);
150 
151     //create paragraphs with ride data
152     var distanceP = document.createElement("P");
153     distanceP.innerHTML = distance;
154     var topSpeedP = document.createElement("P");
155     topSpeedP.innerHTML = topSpeed;
156     var avgSpeedP = document.createElement("P");
157     avgSpeedP.innerHTML = avgSpeed;
158     var startTimeP = document.createElement("P");
159     startTimeP.innerHTML = startTime;
160     var endTimeP = document.createElement("P");
161     endTimeP.innerHTML = endTime;
162     var dateP = document.createElement("P");
163     dateP.innerHTML = date;
164 
165     //create ride deletion button
166     var deleteButton = document.createElement("BUTTON");
167     deleteButton.innerHTML = "Slet tur";
168     deleteButton.onclick = function () { deleteRide(rideNum); };
169 
170     //create div for holding ride content
171     var rideContent = document.createElement("div");
172     rideContent.className = "rideContent";
173     rideContent.id = SERVER_SENT_RIDES;
174 
175     //add ride data to div
176     rideContent.appendChild(distanceP);
177     rideContent.appendChild(topSpeedP);
178     rideContent.appendChild(avgSpeedP);
179     rideContent.appendChild(startTimeP);
180     rideContent.appendChild(endTimeP);
181     rideContent.appendChild(dateP);
182     rideContent.appendChild(deleteButton);
183 
184     //create collapsible for ride
185     rideContainer = document.createElement("button");
186     rideContainer.className = "collapsible";
187     rideContainer.innerHTML = "Tur " + rideNum;
188     rideContainer.onclick = function () { collapsibleFunc(SERVER_SENT_RIDES); };
189 
190     //add div to collapsible
191     rideContainer.appendChild(rideContent);
192     document.getElementById("rideDataContent").appendChild(rideContainer);//put everything into the page
193 }

195 /**
196  * Sørger for at folde en tur ind og ud, når brugeren trykkes på dens collapsible
197  * @param {string} rideContentID Den tur, der er trykket på 
198  */
199 function collapsibleFunc(rideContentID) {
200     content = document.getElementById(rideContentID);
201     if (content.style.display === "block") {
202         content.style.display = "none";
203     } else {
204         content.style.display = "block";
205     }
206 }

234 /**
235  * Tager imod det antal sekunder der er tilbage af timeoutet og viser det på brugerfladen
236  * @param {string} PIN_VALIDATE_RETURNED_TIMEOUT Det antal sekunder, der er tilbage af timeout
237  */
238 function timeOut(PIN_VALIDATE_RETURNED_TIMEOUT) {
239     var timeOutValue = PIN_VALIDATE_RETURNED_TIMEOUT.substr(2)
240     switch (timeOutValue) {
241         case "0":
242             output = "Prøv igen";
243             document.getElementById("wrongPinPrompt").style.display = "inline";
244             document.getElementById("wrongPinPrompt").innerHTML = output;
245             break;
246         case "1":
247             output = "Vent " + PIN_VALIDATE_RETURNED_TIMEOUT.substr(2, 3) + " sekund";
248             document.getElementById("wrongPinPrompt").style.display = "inline";
249             document.getElementById("wrongPinPrompt").innerHTML = output;
250             break;
251         default:
252             output = "Vent " + PIN_VALIDATE_RETURNED_TIMEOUT.substr(2, 3) + " sekunder";
253             document.getElementById("wrongPinPrompt").style.display = "inline";
254             document.getElementById("wrongPinPrompt").innerHTML = output;
255             break;
256     }
257 }

276 /**
277  * Sender en indtastet PIN, der skal valideres, til serveren sammen med eventet USER_UNLOCK_TRY
278  */
279 function validatePin() {
280     var pin = document.getElementById("pin_input").value;
281     if (pin.length == 4) {
282         doSend(eventEnums.USER_UNLOCK_TRY + pin);
283         document.getElementById("pin_input").value = "";
284     }
